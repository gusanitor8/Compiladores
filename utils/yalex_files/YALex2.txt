let delim = [' ''\t''\n']
let ws = delim+
let letra = ['a'-'z''A'-'Z']
let digit = ['0'-'9']
let digits = digit+
let id = (letra)(letra|digit)*
let number = digits('.'digits)?(('E'('_'|'-')?)digits)?

rule tokens =
  | id        { 'ID' }
  | number    { 'NUMBER' }
  | '+'       { 'PLUS' }
  | '-'       { 'MINUS' }
  | '*'       { 'TIMES' }
  | '/'       { 'DIV' }
  | '('       { 'LPAREN' }
  | ')'       { 'RPAREN' }
